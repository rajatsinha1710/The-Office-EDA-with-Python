```Python
text = ""
for words in dataset["About"].str.split(" "):
    for word in words:
        text = text + ' ' + word
wordcloud = WordCloud(width=900, height=400, background_color="#0f4c5c").generate(text)
plt.figure(figsize=(20,10))
plt.imshow(wordcloud, interpolation='bilinear')
plt.axis("off")
plt.margins(x=0, y=0)
plt.show()
```
